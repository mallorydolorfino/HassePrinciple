\chapter{Hilbert symbol}

\section{Hilbert symbol over a field}

\begin{definition}
	\label{def:Hilbert_symbol}
	\lean{HilbertSymbol}
	\leanok
	Let $k$ be a field and $a, b \in k$. If both $a,b$ are nonzero, the \emph{Hilbert symbol} $(a, b)_k$ is defined as
	follows: if there exist $x, y, z \in k$, not all zero, such that $z^2 = ax^2 + by^2$,
	then $(a, b)_k = 1$; otherwise, $(a, b)_k = -1$. If one of $a,b$ is zero, we set $(a, b)_k = 0$.
\end{definition}


\subsection{Basic properties of the Hilbert symbol}

The Hilbert symbols has the following properties, useful for computing it in practice.

\begin{lemma}
	\label{lem:well_defined_up_to_squares}
	\lean{HilbertSymbol.well_defined_up_to_squares}
	\uses{def:Hilbert_symbol}
	\leanok
	Let $a, b, a', b' \in k^\times$. Then $(a,b)_k = (a a'^2, b b'^2)_k$.
\end{lemma}

\begin{proof}
	Let $x, y, z \in k$ be such that $z^2 = a a'^2 x^2 + b b'^2 y^2$. Then, $xa', yb', z$ are not all
	zero and satisfy $z^2 = ax^2 + by^2$. Conversely, if $x, y, z \in k$ are such that
	$z^2 = ax^2 + by^2$, then $x/a', y/b', z$ are not all zero and satisfy
	$z^2 = a a'^2 x^2 + b b'^2 y^2$.
\end{proof}

The following definition and lemma use \code{QuadraticAlgebra} from \mathlib.

\begin{definition}
	\label{def:kb}
	\lean{HilbertSymbol.kb}
	\leanok
	For $b \in k^\times$, we denote by $k(\sqrt{b})$ the field extension of $k$ obtained by adjoining
	a square root of $b$.
\end{definition}

\begin{lemma}
	\label{lem:eq_one_iff}
	\lean{HilbertSymbol.eq_one_iff}
	\uses{def:Hilbert_symbol, def:kb}
	\leanok
	Let $a, b \in k^\times$. Then $(a, b)_k = 1$ if and only if there exists $t \in k(\sqrt{b})$ such
	that $a = N_{k(\sqrt{b})/k}(t)$.
\end{lemma}

\begin{proof}
	\begin{itemize}
		\item Case $b$ is a square in $k$. Then $k(\sqrt{b}) = k$, and the norm map is the identity, so
		$a$ is always the norm of itself. Let $c \in k^\times$ be such that $c^2 = b$.
		Then $c^2 = a \cdot 0^2 + b \cdot 1^2$, so $(a,b)_k=1$.
		\item Case $b$ is not a square in $k$. Then $k(\sqrt{b})$ is a quadratic extension of $k$, and
		the norm map is given by $N_{k(\sqrt{b})/k}(p + q \sqrt{b}) = p^2 - b q^2$.
		Suppose $a$ is a norm, so it satisfies $a = p^2 - b q^2$ for some $p, q \in k$.
		Then $1, q, p$ are not all zero and satisfy $ p^2 = a \cdot 1^2 + b \cdot q^2$,
		so $(a, b)_k = 1$. Conversely, if $x, y, z$ are not all zero and satisfy $z^2 = ax^2 + by^2$,
		then $x=0$, since otherwise $b$ would be a square in $k$, which contradicts our assumption.
		Then $x \ne 0$, and we have $a = (z/x)^2 - b (y/x)^2$, so $a$ is a norm.
	\end{itemize}
\end{proof}

\begin{proposition}
	\label{prop:hilbert_symbol_properties}
	\lean{HilbertSymbol.comm, HilbertSymbol.one_of_square, HilbertSymbol.one_of_neg_self,
	HilbertSymbol.one_of_one_minus_self, HilbertSymbol.of_mul, HilbertSymbol.of_neg_mul,
	HilbertSymbol.of_minus_self_mul}
	\uses{def:Hilbert_symbol}
	\leanok
	The Hilbert symbol statisfies the following formulas:
	\begin{itemize}
		\item For $a, b \in k$, we have $(a, b)_k = (b, a)_k$.
		\item Let $a, c \in k$, both nonzero. Then $(a, c^2)_k = 1$.
		\item For $a \in k$, both nonzero, we have $(a, -a)_k = 1$.
		\item Let $a \in k$ with $a \ne 0, 1$. Then $(a, 1 - a)_k = 1$.
		\item Let $a, a', b \in k$, all nonzero and assume that $(a, b)_k = 1$. Then $(a a', b)_k = (a', b)_k$.
		\item Let $a, b \in k^\times$. Then $(a, -ab)_k = (a, b)_k$.
		\item Let $a, b \in k^\times$ with $a \ne 1$. Then $(a, (1-a)b)_k = (a, b)_k$.
	\end{itemize}
\end{proposition}

\begin{proof}
	\uses{def:Hilbert_symbol, def:kb, lem:eq_one_iff}
	Unless stated otherwise, we can always assume that $a$ and $b$ are nonzero, since otherwisethe statements are trivial.
	\begin{itemize}
		\item If $x,y,z$ is a solution to $z^2 = ax^2 + by^2$, then $y,x,z$ is a solution to
	$z^2 = bx^2 + ay^2$.
		\item In this case, $c^2 = a \cdot 0^2 + b \cdot 1^2$, so $(a,b)_k=1$.
		N.B. Alternatively: the same argument as in the first case of the proof of Lemma~\ref{lem:eq_one_iff} applies.
		\item In this case, $0 = a \cdot 1^2 + (-a) \cdot 1^2$.
		\item In this case, $1 = a \cdot 1^2 + (1 - a) \cdot 1^2$.
		\item We know, by Lemma~\ref{lem:eq_one_iff}, that there exists $t \in k(\sqrt{b})$ such that
		$a = N_{k(\sqrt{b})/k}(t)$. If $(a',b)_k=1$, then $a'$ is the norm of some $t' \in k(\sqrt{b})$,
		so $aa' = N_{k(\sqrt{b})/k}(t t')$, so $(aa', b)_k = 1$. Conversely, if $(aa', b)_k = 1$, then
		$aa' = N_{k(\sqrt{b})/k}(t'')$ for some $t'' \in k(\sqrt{b})$, so
		$a' = N_{k(\sqrt{b})/k}(t''/t)$, hence $(a', b)_k = 1$.
		\item Follows from the previous steps, since $(a, -a)_k = 1$.
		\item Follows from the previous steps, since $(a, 1 - a)_k = 1$.
	\end{itemize}
\end{proof}
\subsection{Local Properties}
We now specialize to the case where $k$ is either $\R$ or $\Q_p$ for some prime $p$. In this case, the Hilbert symbol can be computed explicitly.

\begin{theorem}
	\label{thm:real}
	\lean{HilbertSymbol.real}
	\uses{def:Hilbert_symbol}
	\leanok
	Let $a, b \in \R^\times$. Then $(a, b)_{\R} = 1$ if and only if $a > 0$ or $b > 0$; otherwise $(a, b)_{\R} = -1$.
\end{theorem}

\begin{proof}
	If $a>0$, then $(1,0,\sqrt{a})$ works. Similarly, if $b>0$, then $(0,1,\sqrt{b})$ works.
	If both $a$ and $b$ are negative, then $ax^2 + by^2 \le 0$ for all $x,y$, so there is no
	nontrivial solution to $z^2 = ax^2 + by^2$.
\end{proof}

\subsection{Global properties of the Hilbert symbol}

This section uses the \code{mathlib} function \code{Nat.nth Nat.Prime}. The main goal is to prove
the product formula for the Hilbert symbol, which states that for every $a, b \in \Q^\times$, we have
\[\prod_{v \in V} (a, b)_v = 1.\]
Here, $V$ is the union of the set of prime numbers and the symbol $\infty$, and for each prime $p$,
the symbol $(a, b)_p$ is the Hilbert symbol computed in $\Q_p$, while $(a, b)_\infty$ is the Hilbert
symbol computed in $\R$. Prior to stating the formula, we need to show that the Hilbert
symbol is $1$ for all but finitely many $v \in V$.

\begin{definition}
	\label{def:at_p}
	\lean{HilbertSymbol.at_p}
	\uses{def:Hilbert_symbol}
	\leanok
Let $p$ be a prime number. For $a, b \in \Q^\times$, we define $(a, b)_p$ as the Hilbert symbol
$(a,b)_{\Q_p}$ computed in $\Q_p$.
\end{definition}

\begin{definition}
	\label{def:at_infty}
	\lean{HilbertSymbol.at_infty}
	\uses{def:Hilbert_symbol}
	\leanok
Let $a, b \in \Q^\times$. We define $(a, b)_\infty$ as the Hilbert symbol $(a,b)_{\R}$ computed in $\R$.
\end{definition}

	\leanok
Let $a, b \in \Q^\times$. Then there exists a finite set $S$ of prime numbers such that
$(a, b)_p = 1$ for all primes $p \notin S$.

In order to state the product formula, it is convenient to introduce the following notation.
\begin{definition}
	\label{def:hilbertSym_fun}
	\lean{HilbertSymbol.hilbertSym_fun}
	\uses{def:Hilbert_symbol, def:at_p, def:at_infty}
	\leanok
Let $a, b \in \Q^\times$. The Hilbert symbol function of the pair $(a,b)$ is
\[
\rm{hilbertSym\_fun}: \N \to \Z^\times
\]
sending each $n$ to the Hilbert symbol of the pair $(a,b)$ at the $n$-th prime.
\end{definition}

By Theorem \ref{thm:almost_all_one}, this function has finite support.

\begin{definition}
	\label{def:hilbertSym_support}
	\lean{HilbertSymbol.hilbertSym_support}
	\uses{def:Hilbert_symbol, def:at_p, def:at_infty, def:hilbertSym_fun}
	\leanok
Let $a, b \in \Q^\times$. The support of the Hilbert symbol function of the pair $(a,b)$ is
\[
\rm{hilbertSym\_support}(a, b) := \{n \in \N \mid \ hilbertSym\_fun(a, b)(n) \neq 1\}.
\]
\end{definition}

\subsection{Approximation and Existence Theorem}


For the statement of the approximation theorem, it is convenient to define the following product
space.

\begin{definition}
	\label{def:prod_over_S}
	\lean{Rat.prod_over_S}
	\leanok
Let $S$ be a finite subset of the set of prime numbers. We denote by $\prod_{p \in S} \Q_p$
the product of the fields $\Q_p$ for $p \in S$. The set $\rm{Prod\_over\_S} (S)$ is defined as the
product of $\R$ and the factors $\Q_p$ for $p \in S$.
\end{definition}

N.B. more precisely, in the Lean code, S is a subset of the natural numbers, and the product is
over the $n$-th prime for $n \in S$.

\begin{definition}
	\label{def:finite_embedding}
	\lean{Rat.finite_embedding}
	\uses{def:prod_over_S}
	\leanok
We define the finite embedding of $\Q$ into $\rm{Prod\_over\_S} (S)$ as the natural diagonal map.
\end{definition}

	\forall i \in I, \forall v \text{ place of } \Q, (x,a_i)_v=e_{i,v}

	\end{itemize}
\end{theorem}
